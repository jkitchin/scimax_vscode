This file generates a dynamic agenda of documentation headings that need review.
Headings marked with ‚ö†Ô∏è are new and need inspection before being marked ‚úÖ.

* Generate Review Agenda

Run this block (C-c C-c) to generate the current review agenda from docs/*.org files.

#+begin_src sh :results output raw
cd /Users/jkitchin/Dropbox/emacs/projects/scimax_vscode/docs

echo "* Review Agenda"
echo ""

total=0
for file in *.org; do
    [ "$file" = "review-agenda.org" ] && continue

    # Find lines with ‚ö†Ô∏è or üëÄ headings
    matches=$(grep -n "^\*.*‚ö†Ô∏è\|^\*.*üëÄ" "$file" 2>/dev/null || true)

    if [ -n "$matches" ]; then
        count=$(echo "$matches" | wc -l | tr -d ' ')
        total=$((total + count))
        echo "** [[file:$file][$file]] ($count items)"
        echo "$matches" | while IFS=: read -r linenum content; do
            # Extract the heading title (remove stars and TODO keyword)
            title=$(echo "$content" | sed 's/^\*\+ //' | sed 's/^‚ö†Ô∏è //' | sed 's/^üëÄ //')
            echo "   - [[file:$file::$linenum][$title]]"
        done
        echo ""
    fi
done

if [ "$total" -eq 0 ]; then
    echo "No items marked with ‚ö†Ô∏è or üëÄ found. All documentation reviewed!"
else
    echo "Total: $total items need review"
fi
